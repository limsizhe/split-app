    <style>
        body { font-family: sans-serif; background: var(--tg-theme-bg-color); color: var(--tg-theme-text-color); padding: 20px; }
        input, select { width: 100%; padding: 10px; margin: 10px 0; border-radius: 8px; border: 1px solid #ccc; }

<body>
    <h3>New Expense</h3>
    <input type="number" id="amount" placeholder="Amount">
    <select id="currency">
        <option value="USD">USD</option>
        <option value="EUR">EUR</option>
        <option value="SGD">SGD</option>
    </select>
    <input type="text" id="desc" placeholder="What was it for?">
    <p>Split with (comma separated names):</p>
    <input type="text" id="who" placeholder="Alex, Sam, Jordan">
    
    <button onclick="sendData()">Add to Sheet</button>

    <script>
        const tg = window.Telegram.WebApp;
        tg.ready();

        function sendData() {
            const data = {
                amount: document.getElementById('amount').value,
                currency: document.getElementById('currency').value,
                description: document.getElementById('desc').value,
                who: document.getElementById('who').value
            };
            // This sends the data as a string to the bot chat
            tg.sendData(JSON.stringify(data));
            tg.close(); // Closes the mini app window
        }
    </script>
